# 한수
n = int(input())

def arithmetic_sequence(x):
  cnt = 0
  if x >= 1000:
    x = 999
  if x < 100:
      cnt = x
  else:
    cnt = 99
    for i in range(100, x+1):
      x_list = list(map(int, str(i)))
      if x_list[0]-x_list[1] == x_list[1]-x_list[2]:
        cnt += 1

  return cnt, x_list

print(arithmetic_sequence(n))

# 제곱수의 합 (43에서 예외 발생 -> 6,2,1,1,1 로 5가 나오는데 5,3,3으로 3번에 가능함)
import math
n = int(input())

def double_num(x):
  cnt = 0
  while x>0:
    cnt += 1
    x -= int(math.sqrt(x))**2
  return cnt

print(double_num(n))

# 제곱수의 합 (다이나믹 프로그래밍)
n = int(input())
dp = [0 for i in range(n+1)]  # dp테이블 초기화

def double_sum(x):
  for i in range(1,x+1):
    dp[i] = i
    for j in range(1,i):
      if j**2>i:  # 제곱수가 i보다 커지면 정지
        # print(j, i)
        break
      if dp[i]>dp[i-j**2]+1:
        # print(dp[i])
        dp[i] = dp[i-j**2]+1
        # print(dp[i])
  
  return dp[x]

print(double_sum(n))

# 괄호 변환 (카카오 코테 1차 문제 / 코드 거의 작성 못함ㅜ)
p = input()

def balance_str(p):
  cnt = 0
  for i in range(len(p)):
    if p[i] == '(':
      cnt += 1
    else:
      cnt -= 1
    if cnt == 0:
      return i

# 올바른 괄호 문자열 확인
def correct_str(p):
  cnt = 0
  for i in p:
    if i == '(':
      cnt += 1
    else:
      if cnt == 0:  # '('보다 ')'가 먼저 나온 경우
        return False
      cnt -= 1
  return True

def print_str(p):
  answer = ''
  if p == '':  # 비어있는 경우
    return answer
  index = balance_str(p)
  u = p[:index+1]
  v = p[index+1:]

  if correct_str(u):
    answer = u + print_str(v)
  else:
    answer = '('  # 없으면 올바른 괄호로 시작하지 않을 때 결과값이 달라짐
    answer += print_str(v)
    answer += ')'
    u = list(u[1:-1])  # 첫번째와 마지막 문자만 삭제하고 u에 저장
    for i in range(len(u)):
      if u[i] == '(':
        u[i] = ')'
      else:
        u[i] = '('
    answer += ''.join(u)
  return answer

print(print_str(p))
