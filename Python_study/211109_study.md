```
# 반복수열

a, p = input().split()
p = int(p)
remain = [a]
i = 0

def mul(x):
  x = int(x)**p
  return x

while True:
  i += 1
  num = sum(map(mul, str(remain[i-1])))
  
  if num in remain:
    break

  remain.append(num)

print(remain.index(num))
```

```
# 5학년은 다니기 싫어요

n, a, b = map(int, input().split())
x = [list(map(int, input().split())) for _ in range(10)]
i = 0

while True:
  n += 1
  for _ in range(x[i][0]):
    if a <= 66:
      a += 1*3
    else:
      continue
  for _ in range(x[i][1]):
    if b <= 130:
      b += a + 1*3
    else:
      continue
  i += 1

  if n == 8:
    break

if a >= 66 and b >= 130:
  print("Nice")
else:
  print("Nae ga wae")
```
