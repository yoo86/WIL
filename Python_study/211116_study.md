# 숫자 게임
import itertools

max_num = 0
score = []
n = int(input())
a = []
for _ in range(n):
  num = list(map(int, input().split()))
  mix = list(itertools.combinations(num,3))
  for i in mix:
    a.append(sum(i)%10)
    a.sort()
  if a[-1] >= max_num:
    max_num = a[-1]
    score.append(max_num)

for j in range(len(score)):
  if score[j] == max_num:
    print(j+1)
    break
```

# 악수 (악수를 하지 않는 경우와 악수를 한번만 할 수 있는 경우)
import itertools

n = int(input())
cnt = 0   # 악수 횟수
person = []   # 앉은 자리

for i in range(1, n+1):   # 1번~n번을 부여
  person.append(i)

case1 = list(itertools.combinations(person,2))

for i in case1:
  case2 = i
  if case2[1] - case2[0] == 1:
    cnt += 1
  
print(cnt+1)
```
